def make_tmpname(basename, name)
  case basename
    when Array
      prefix, suffix = *basename
    else
      prefix, suffix = basename, ''
  end

  "#{prefix}#{Time.now.strftime('%Y%m%d')}-#{$$}-#{rand(0x100000000).to_s(36)}-#{name}#{suffix}"
end
